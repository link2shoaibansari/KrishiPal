    function toggleDarkMode() {
      const root = document.documentElement;
      if (root.style.getPropertyValue('--bg-color') === '#121212') {
        root.style.setProperty('--bg-color', '#e8f5e9');
        root.style.setProperty('--text-color', '#2e7d32');
        root.style.setProperty('--card-bg', 'rgba(255, 255, 255, 0.6)');
        root.style.setProperty('--accent-color', '#66bb6a');
      } else {
        root.style.setProperty('--bg-color', '#121212');
        root.style.setProperty('--text-color', '#ffffff');
        root.style.setProperty('--card-bg', 'rgba(33, 33, 33, 0.7)');
        root.style.setProperty('--accent-color', '#81c784');
      }
    }

    document.getElementById('pirToggle').addEventListener('change', function () {
      const state = this.checked ? 'activated' : 'deactivated';
      alert(`PIR Sensor ${state}`);
    });

    document.getElementById('pumpToggle').addEventListener('change', function () {
      const state = this.checked ? 'ON' : 'OFF';
      alert(`Pump turned ${state}`);
    });
